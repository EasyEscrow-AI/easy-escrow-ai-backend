**Step 4: Asset Verification and Safety Checks**
```javascript
// test/production-e2e/asset-verification.js
async function verifyUserAssets() {
  // Verify USDC balance
  const usdcBalance = await getTokenBalance(
    config.wallets.sender.address,
    config.testAssets.usdc.mint
  );
  
  if (usdcBalance < config.testAssets.usdc.requiredAmount) {
    throw new Error(`Insufficient USDC balance. Required: ${config.testAssets.usdc.requiredAmount}, Available: ${usdcBalance}`);
  }

  // Verify NFT ownership
  const nftOwnership = await verifyNFTOwnership(
    config.wallets.sender.address,
    config.testAssets.nft.requiredCollection
  );
  
  if (!nftOwnership.isValid) {
    throw new Error('Required NFT not found in sender wallet');
  }

  // Verify SOL balance for transaction fees
  const solBalance = await getSolBalance(config.wallets.sender.address);
  if (solBalance < 0.1) {
    throw new Error('Insufficient SOL for transaction fees');
  }
}
```
