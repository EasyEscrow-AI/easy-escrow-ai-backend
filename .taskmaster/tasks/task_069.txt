**Step 3: Configure DigitalOcean App Platform Secrets**
```bash
# Update staging app spec to use .env.staging or DO secrets
doctl apps create-deployment <staging-app-id> --spec staging-app.yaml
```

Create staging-app.yaml with proper naming:
```yaml
name: easyescrow-staging
services:
- name: api
  environment_slug: node-js
  github:
    repo: your-org/easyescrow
    branch: staging
  envs:
  - key: NODE_ENV
    value: staging
  - key: SOLANA_NETWORK
    value: devnet
  - key: DEVNET_STAGING_PROGRAM_ID
    type: SECRET
    value: <encrypted_staging_program_id>
  - key: DATABASE_URL
    type: SECRET
    value: <encrypted_staging_db_url>
  - key: REDIS_URL
    type: SECRET
    value: <encrypted_staging_redis_url>
  - key: DEVNET_STAGING_SENDER_PRIVATE_KEY
    type: SECRET
    value: <encrypted_staging_sender_key>
```
