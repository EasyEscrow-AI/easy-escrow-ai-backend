Create a complete CI/CD pipeline for the STAGING environment using GitHub Actions:

**Stage 1: Build (On merge to staging branch or tag)**
```yaml
# .github/workflows/build-staging.yml
name: Build for STAGING

on:
  push:
    branches: [staging]
    tags: ['staging-*']

jobs:
  build:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      
      - name: Pin Solana CLI
        run: |
          sh -c "$(curl -sSfL https://release.solana.com/v1.18.x/install)"
          echo "$HOME/.local/share/solana/install/active_release/bin" >> $GITHUB_PATH
      
      - name: Pin Rust toolchain
        run: rustup show  # Reads rust-toolchain.toml
      
      - name: Build program
        run: anchor build
      
      - name: Generate checksums
        run: |
          shasum -a 256 target/deploy/escrow.so > target/deploy/escrow.so.sha256
          shasum -a 256 target/idl/escrow.json > target/idl/escrow.json.sha256
      
      - name: Run unit tests
        run: anchor test --skip-local-validator
      
      - name: Upload artifacts
        uses: actions/upload-artifact@v3
        with:
          name: program-artifacts
          path: |
            target/deploy/escrow.so
            target/deploy/escrow.so.sha256
            target/idl/escrow.json
            target/idl/escrow.json.sha256
```

**Stage 2: Deploy to STAGING (Artifact promotion)**
```yaml
# .github/workflows/deploy-staging.yml
name: Deploy to STAGING

on:
  workflow_run:
    workflows: ["Build for STAGING"]
    types: [completed]

jobs:
  deploy:
    runs-on: ubuntu-latest
    if: ${{ github.event.workflow_run.conclusion == 'success' }}
    environment: staging  # Requires manual approval in GitHub
    
    steps:
      - uses: actions/checkout@v3
      
      - name: Download artifacts
        uses: actions/download-artifact@v3
        with:
          name: program-artifacts
          path: target/deploy
      
      - name: Setup Solana CLI
        run: |
          sh -c "$(curl -sSfL https://release.solana.com/v1.18.x/install)"
          echo "$HOME/.local/share/solana/install/active_release/bin" >> $GITHUB_PATH
      
      - name: Configure staging deployer
        env:
          STAGING_DEPLOYER_KEY: ${{ secrets.STAGING_DEPLOYER_KEYPAIR }}
        run: |
          echo "$STAGING_DEPLOYER_KEY" > /tmp/staging-deployer.json
          chmod 600 /tmp/staging-deployer.json
      
      - name: Deploy to STAGING
        env:
          STAGING_RPC_URL: ${{ secrets.STAGING_RPC_URL }}
          STAGING_PROGRAM_ID: ${{ secrets.STAGING_PROGRAM_ID }}
        run: |
          anchor deploy \
            -C Anchor.staging.toml \
            --provider.cluster devnet \
            --provider.wallet /tmp/staging-deployer.json
      
      - name: Update IDL
        run: |
          anchor idl upgrade $STAGING_PROGRAM_ID \
            target/idl/escrow.json \
            -C Anchor.staging.toml \
            --provider.cluster devnet
      
      - name: Deploy backend to DO App Platform
        env:
          DIGITALOCEAN_ACCESS_TOKEN: ${{ secrets.DIGITAL_OCEAN_API_KEY }}
        run: |
          doctl apps create-deployment $STAGING_APP_ID \
            --wait \
            --timeout 10m
      
      - name: Run smoke tests
        run: npm run test:staging:smoke
      
      - name: Notify deployment
        uses: 8398a7/action-slack@v3
        with:
          status: ${{ job.status }}
          text: 'STAGING deployment completed'
          webhook_url: ${{ secrets.SLACK_WEBHOOK }}
```

**Rollback Procedure (CI-based)**
```yaml
# .github/workflows/rollback-staging.yml
name: Rollback STAGING

on:
  workflow_dispatch:
    inputs:
      deployment_id:
        description: 'Deployment ID to rollback to'
        required: true

jobs:
  rollback:
    runs-on: ubuntu-latest
    steps:
      - name: Rollback DO App
        run: |
          doctl apps get-deployment ${{ github.event.inputs.deployment_id }}
          # Redeploy that specific deployment
```

const axios = require('axios');
const { Connection } = require('@solana/web3.js');

async function runSmokeTests() {
  // Health check

runSmokeTests().catch(err => {
  console.error('Smoke tests failed:', err);

});

2. **Migration Script**
Create a database migration script that can be executed via `npm run staging:migrate`:
```javascript
// scripts/migrate-staging.js
const { Pool } = require('pg');
const fs = require('fs');
const path = require('path');

async function runMigrations() {
  const pool = new Pool({
    connectionString: process.env.STAGING_DATABASE_URL
  });
  
  // Get migration files
  const migrationDir = path.join(__dirname, '../migrations');
  const files = fs.readdirSync(migrationDir).sort();
  
  // Run each migration
  for (const file of files) {
    const migration = fs.readFileSync(path.join(migrationDir, file), 'utf8');
    await pool.query(migration);
    console.log(`Applied migration: ${file}`);
  }
  
  await pool.end();
}

runMigrations().catch(err => {
  console.error('Migration failed:', err);
  process.exit(1);
});
```
