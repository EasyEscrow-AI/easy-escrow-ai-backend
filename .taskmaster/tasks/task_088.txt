**Step 1: Create Pre-Deployment Test Runner**
```typescript
// tests/pre-deployment/staging-validation.ts
import { execSync } from 'child_process';
import { StagingSmokeTests } from '../smoke/staging-smoke-tests';

interface ValidationResult {
  testSuite: string;
  passed: boolean;
  duration: number;
  failedTests?: string[];
  error?: string;
}

class PreDeploymentValidator {
  async runFullValidation(): Promise<ValidationResult[]> {
    const results: ValidationResult[] = [];
    
    // Run smoke tests first
    console.log('🔍 Running smoke tests...');
    const smokeResult = await this.runSmokeTests();
    results.push(smokeResult);
    
    if (!smokeResult.passed) {
      throw new Error('Smoke tests failed - aborting validation');
    }
    
    // Run E2E tests
    console.log('🧪 Running E2E tests...');
    const e2eResult = await this.runE2ETests();
    results.push(e2eResult);
    
    return results;
  }
  
  private async runSmokeTests(): Promise<ValidationResult> {
    try {
      const smokeTests = new StagingSmokeTests();
      const results = await smokeTests.runAllTests();
      const failedTests = results.filter(r => !r.passed);
      
      return {
        testSuite: 'Smoke Tests',
        passed: failedTests.length === 0,
        duration: results.reduce((sum, r) => sum + r.duration, 0),
        failedTests: failedTests.map(t => t.name)
      };
    } catch (error) {
      return {
        testSuite: 'Smoke Tests',
        passed: false,
        duration: 0,
        error: error.message
      };
    }
  }
  
  private async runE2ETests(): Promise<ValidationResult> {
    try {
      const startTime = Date.now();
      execSync('npm run test:e2e:staging', { 
        stdio: 'inherit',
        env: { ...process.env, NODE_ENV: 'staging' }
      });
      
      return {
        testSuite: 'E2E Tests',
        passed: true,
        duration: Date.now() - startTime
      };
    } catch (error) {
      return {
        testSuite: 'E2E Tests',
        passed: false,
        duration: Date.now() - startTime,
        error: error.message
      };
    }
  }
}
```

// Generate detailed validation report
class ValidationReporter {
  generateReport(results: ValidationResult[]): string {
    const totalTests = results.length;
    const passedTests = results.filter(r => r.passed).length;
    const totalDuration = results.reduce((sum, r) => sum + r.duration, 0);
    
    return `
# Pre-Deployment Validation Report
Generated: ${new Date().toISOString()}

## Summary
- Total Test Suites: ${totalTests}
- Passed: ${passedTests}
- Failed: ${totalTests - passedTests}
- Total Duration: ${totalDuration}ms

## Test Results
${results.map(r => this.formatTestResult(r)).join('\n')}

## Production Readiness: ${passedTests === totalTests ? '✅ READY' : '❌ NOT READY'}
    `;
  }
}
