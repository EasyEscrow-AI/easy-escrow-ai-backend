const dbTracker = {
  async trackQueryPerformance(query: string, startTime: number) {
    const duration = Date.now() - startTime;
    if (duration > 1000) { // Slow query threshold
      await resourceTracker.trackSlowQuery(query, duration);
    }
    await resourceTracker.trackDatabaseQuery(query, duration);
